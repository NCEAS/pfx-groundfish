
model {
  	for(i in 1:N.obs){
  		y[i] ~ dpois(exp(lambda[i]))
   		eta[i] ~ dnorm(0,1/sigma2)
 	}
   	lambda <- X %*% betas + eta


	### Derived Quantities	
		p[1] <- exp(betas[1])
	for(i in 2:N.cov){
		p[i]	<-	exp(betas[1] + betas[i])	
	}
	
	for(i in 1:N.cov){
		P[i]	<-	p[i] / sum(p)
	}	
  	
  	
  	### Priors
  	for(j in 1:N.cov){
	 	betas[j] ~ dnorm(0,0.001) 
    }
    sigma2 ~ dgamma(0.001,0.001)

}
